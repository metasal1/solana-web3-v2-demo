---
title: Get Programs
description: Get all the programs currently on the Solana blockchain
---

# New Way
```tsx
import { address, createSolanaRpc } from "@solana/web3.js";

const rpc = createSolanaRpc('http://127.0.0.1:8899');

const PROGRAM_ID = 'Qi4ioHyQKMx7LMFxDAtVz9BJAaofvSFZWSkixBwyWDD';

const getProgramInfo = async () => {

    const programInfo = await rpc.getAccountInfo(address(PROGRAM_ID)).send()
    console.log(`
      Program Info:
      - Program ID: ${PROGRAM_ID}
      - Owner: ${programInfo.value?.owner}
      - Data Length: ${programInfo.value?.data.length} bytes
      - Lamports: ${programInfo.value?.lamports}
      - Executable: ${programInfo.value?.executable}
    `);
};

getProgramInfo();
```
# Old Way
```tsx
import { Connection, PublicKey } from '@solana/web3.js';

const PROGRAM_ID = 'Qi4ioHyQKMx7LMFxDAtVz9BJAaofvSFZWSkixBwyWDD';

const getProgramInfo = async () => {
    const connection = new Connection(`http://localhost:8899`, 'confirmed');

    try {

        const publicKey = new PublicKey(PROGRAM_ID);

        const accountInfo = await connection.getAccountInfo(publicKey);

        console.log(`
      Program Info:
      - Program ID: ${publicKey.toBase58()}
      - Owner: ${accountInfo.owner.toBase58()}
      - Data Length: ${accountInfo.data.length} bytes
      - Lamports: ${accountInfo.lamports}
      - Executable: ${accountInfo.executable}
    `);
    } catch (error) {
        console.error('Error fetching program info:', error.message);
    }
};

getProgramInfo();

```
